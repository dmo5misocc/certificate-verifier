  <script>

    async function verifyCert() {

      const certId = document.getElementById("certId").value.trim();

      const resultDiv = document.getElementById("result");

      resultDiv.innerHTML = "Checking…";


      try {

        const res = await fetch(`https://script.google.com/macros/s/AKfycbwjAkxT3JNtCwWpjH6lvTX_W3iHIioP6GhrE-hpnZJZlrDddH-5i_NFCMfwtYviQWN2/exec?certId=${certId}`);

        const html = await res.text();

        if (html.includes("Invalid Certificate")) {

          resultDiv.innerHTML = `<p class="invalid">❌ Invalid Certificate</p>`;

        } else {

          resultDiv.innerHTML = html;

        }

      } catch (e) {

        resultDiv.innerHTML = `<p class="invalid">❌ Error verifying certificate.</p>`;

      }

    }


    // Auto-fill from URL

    const params = new URLSearchParams(window.location.search);

    const certId = params.get("certId");

    if (certId) {

      document.getElementById("certId").value = certId;

      verifyCert();

    }

  </script>
